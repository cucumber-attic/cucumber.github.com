def matched_arguments(step_name)
  match = @regexp.match(step_name)
  if (match)
    n = 0
    match.captures.map do |val|
      n += 1
      start = match.offset(n)[0]
      Java::GherkinFormatter::Argument.new(start, val)
    end
  else
    nil
  end
end
