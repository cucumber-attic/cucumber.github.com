def end_with?(str) #:nodoc:

  tail = self[-str.length, str.length]
  tail == str
end

def indent(n)
  if n >= 0
    gsub(/^/, ' ' * n)
  else
    gsub(/^ {0,#{-n}}/, "")
  end
end
