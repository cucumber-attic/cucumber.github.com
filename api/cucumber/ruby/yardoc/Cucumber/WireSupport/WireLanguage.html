def snippet_text(code_keyword, step_name, multiline_arg_class, snippet_type)
  snippets = @connections.map do |remote|
    remote.snippet_text(code_keyword, step_name, multiline_arg_class.to_s)
  end
  snippets.flatten.join("\n")
end
