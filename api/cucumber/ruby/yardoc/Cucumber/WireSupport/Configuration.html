def initialize(args)
  @host = args['host']
  @port = args['port']
  @unix = args['unix'] if RUBY_PLATFORM !~ /mingw|mswin/
  @timeouts = DEFAULT_TIMEOUTS.merge(args['timeout'] || {})
end

def self.from_file(wire_file)
  settings = YAML.load(ERB.new(File.read(wire_file)).result)
  new(settings)
end
