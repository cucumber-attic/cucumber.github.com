def accept(visitor)
  return if Cucumber.wants_to_quit
  init
  cells_rows.each_with_index do |row, n|
    if(visitor.configuration.expand?)
      row.accept(visitor)
    else
      visitor.visit_table_row(row)
    end
  end
  nil
end
