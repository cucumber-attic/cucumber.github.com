def cmd
  if use_bundler
    [Cucumber::RUBY_BINARY, '-S', 'bundle', 'exec', 'rcov', @rcov_opts,
     quoted_binary(@cucumber_bin), '--', @cucumber_opts, @feature_files].flatten
  else
    [Cucumber::RUBY_BINARY, '-I', load_path(@libs), '-S', 'rcov', @rcov_opts,
     quoted_binary(@cucumber_bin), '--', @cucumber_opts, @feature_files].flatten
  end
end
