def result(background, language, feature_tags)
  scenario_outline = Ast::ScenarioOutline.new(
    language,
    location,
    background,
    comment,
    tags,
    feature_tags,
    node.keyword,
    node.name,
    node.description,
    steps(language),
    examples_sections
  )
  scenario_outline.gherkin_statement(node)
  scenario_outline
end
